**json 处理**

需要引入"encoding/json"包

**json解析到数组**

```
jsonStr := "{\"a\":\"test\",\"b\":\"testb\"}"
var dat map[string]string
err := json.Unmarshal([]byte(jsonStr), &dat)
if err == nil {
   fmt.Println(dat)
}

结果：
map[a:test b:testb]
```

**json 解析到结构体**

```
package main

import (
   "encoding/json"
   "fmt"
)

type Person struct {
   //结构也是一种类型
   Name string //定义struct的属性
   Age  int
}

func main() {   
   var person Person
   jsonStr := "{\"Name\":\"test\",\"Age\":19}"
   if err := json.Unmarshal([]byte(jsonStr), &person); err == nil {
      fmt.Println(person)
   }
}
结果：
{test 19}
```
